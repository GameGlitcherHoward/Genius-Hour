* { margin: 0; padding: 0; box-sizing: border-box; } body { display: flex; justify-content: center; align-items: center; height: 100vh; background-color: #f0f0f0; } #gameCanvasContainer { position: relative; } canvas { border: 2px solid black; background-color: #87CEEB; /* Sky blue for the paintball field */ } const canvas = document.getElementById('gameCanvas'); const ctx = canvas.getContext('2d'); // Set canvas dimensions canvas.width = 800; canvas.height = 600; // Player object const player = { x: canvas.width / 2, y: canvas.height - 50, width: 30, height: 30, speed: 5, color: 'blue' }; // Paintball object const paintballs = []; const paintballRadius = 5; const paintballSpeed = 7; // Event listeners for controlling the player let keys = {}; document.addEventListener('keydown', (e) => { keys[e.key] = true; }); document.addEventListener('keyup', (e) => { keys[e.key] = false; }); // Shooting paintballs document.addEventListener('click', () => { const paintball = { x: player.x + player.width / 2 - paintballRadius, y: player.y, radius: paintballRadius, speed: paintballSpeed }; paintballs.push(paintball); }); // Update player position based on keys function movePlayer() { if (keys['ArrowLeft'] && player.x > 0) { player.x -= player.speed; } if (keys['ArrowRight'] && player.x + player.width < canvas.width) { player.x += player.speed; } if (keys['ArrowUp'] && player.y > 0) { player.y -= player.speed; } if (keys['ArrowDown'] && player.y + player.height < canvas.height) { player.y += player.speed; } } // Update paintballs function movePaintballs() { for (let i = 0; i < paintballs.length; i++) { paintballs[i].y -= paintballs[i].speed; if (paintballs[i].y < 0) { paintballs.splice(i, 1); // Remove paintball if it goes off-screen i--; } } } // Draw the player and paintballs function draw() { ctx.clearRect(0, 0, canvas.width, canvas.height); // Clear the canvas // Draw player ctx.fillStyle = player.color; ctx.fillRect(player.x, player.y, player.width, player.height); // Draw paintballs ctx.fillStyle = 'red'; for (let i = 0; i < paintballs.length; i++) { ctx.beginPath(); ctx.arc(paintballs[i].x, paintballs[i].y, paintballs[i].radius, 0, Math.PI * 2); ctx.fill(); } } // Main game loop function gameLoop() { movePlayer(); movePaintballs(); draw(); requestAnimationFrame(gameLoop); // Repeat the loop } // Start the game gameLoop();
